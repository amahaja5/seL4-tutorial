\apidoc
{ia32_irq_control_getioapic}
{IRQ Control - GetIOAPIC}
{Create an IRQ handler capability for an interrupt from an IOAPIC}
{static inline int seL4\_IRQControl\_GetIOAPIC }
{
\param{seL4\_IRQControl}{\_service}{An IRQControl capability. This gives you the authority to make this call.}
\param{int}{irq}{The IRQ that you want this capability to handle.}
\param{seL4\_CNode}{root}{\destcspacedesc}
\param{seL4\_Word}{index}{\destindexdesc}
\param{uint8\_t}{depth}{\destdepthdesc}
\param{seL4\_Word}{ioapic}{Zero based index of IOAPIC to get interrupt from, ordered the same as in ACPI tables}
\param{seL4\_Word}{pin}{IOAPIC pin that generates the interrupt}
\param{seL4\_Word}{level}{Indicates whether the IOAPIC should be programmed to treat this interrupt as level triggered}
\param{seL4\_Word}{polarity}{Indicates whether the IOAPIC should be programmed to treat this interrupt as high or low triggered}
\param{seL4\_Word}{vector}{CPU vector to deliver the interrupt to}
}
{\errorenumdesc}
